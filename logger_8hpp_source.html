155 template <typename T> void Logger::log_constr(T *inst, std::vector<rttr::variant> args)
156 {
157  std::unique_lock<std::mutex> lk(mut);
158  for (auto &a : args) {
159  // this will rewove shared/weak/unique ptrs
160  if (a.get_type().is_wrapper()) {
161  a = a.extract_wrapped_value();
162  }
163  }
164  std::string class_name = rttr::type::get<T>().get_name().to_string();
165  constr[class_name].push_back({inst, std::move(args)});
166  operations.emplace_back(ConstrId{class_name, constr[class_name].size() - 1});
167 }

169 template <typename T> void Logger::log(T *inst, std::string fctName, std::vector<rttr::variant> args)
170 {
171  std::unique_lock<std::mutex> lk(mut);
172  for (auto &a : args) {
173  // this will rewove shared/weak/unique ptrs
174  if (a.get_type().is_wrapper()) {
175  a = a.extract_wrapped_value();
176  }
177  }
178  std::string class_name = rttr::type::get<T>().get_name().to_string();
179  invoks.push_back({inst, std::move(fctName), std::move(args), rttr::variant()});
180  operations.emplace_back(InvokId{invoks.size() - 1});
181  result_awaiting = invoks.size() - 1;
182 }

184 template <typename T> size_t Logger::get_id_from_ptr(T *ptr)
185 {
186  const std::string class_name = rttr::type::get<T>().get_name().to_string();
187  for (size_t i = 0; i < constr.at(class_name).size(); ++i) {
188  if (constr.at(class_name)[i].first.convert<T *>() == ptr) {
189  return i;
190  }
191  }
192  std::cerr << "Error: ptr of type " << class_name << " not found" << std::endl;
193  return INT_MAX;
194 }
