25 /* This file contains a collection of macros that can be used in model related classes.
26  The class only needs to have the following members:
27  - For Push_undo : std::weak_ptr<DocUndoStack> m_undoStack; this is a pointer to the undoStack
28  - For Update_undo_redo: mutable QReadWriteLock m_lock; This is a lock that ensures safety in case of concurrent access. Note that the mutex must be
29  recursive.
30 
31  See for example TimelineModel.
32 
33  Note that there also exists a version of update_undo_redo without the need for a lock (but prefer the mutex version where applicable)
34 */

36 /* This convenience macro adds lock/unlock ability to a given lambda function
37  Note that it is automatically called when you push the lambda so you shouldn't have
38  to call it directly yourself
39 */
40 #define LOCK_IN_LAMBDA(lambda) \
41  lambda = [this, lambda]() { \
42  m_lock.lockForWrite(); \
43  bool res_lambda = lambda(); \
44  m_lock.unlock(); \
45  return res_lambda; \
46  };

48 /*This convenience macro locks the mutex for reading.
49 Note that it might happen that a thread is executing a write operation that requires
50 reading a Read-protected property. In that case, we try to write lock it first (this will be granted since the lock is recursive)
51 */
52 #define READ_LOCK() \
53  std::unique_ptr<QReadLocker> rlocker(new QReadLocker(nullptr)); \
54  std::unique_ptr<QWriteLocker> wlocker(new QWriteLocker(nullptr)); \
55  if (m_lock.tryLockForWrite()) { \
56  /*we yield ownership of the lock to the WriteLocker*/ \
57  m_lock.unlock(); \
58  wlocker.reset(new QWriteLocker(&m_lock)); \
59  } else { \
60  rlocker.reset(new QReadLocker(&m_lock)); \
61  }
