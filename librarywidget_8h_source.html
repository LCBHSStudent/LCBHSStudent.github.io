75  QSize sizeHint(const QStyleOptionViewItem &option, const QModelIndex &index) const override
76  {
77  QSize hint = QStyledItemDelegate::sizeHint(option, index);
78  QString text = index.data(Qt::UserRole + 1).toString();
79  QRectF r = option.rect;

90  void paint(QPainter *painter, const QStyleOptionViewItem &option, const QModelIndex &index) const override
91  {
92  if (index.column() == 0) {
93  QRect r1 = option.rect;
94  painter->save();
95  painter->setClipRect(r1);
96  QStyleOptionViewItem opt(option);
97  initStyleOption(&opt, index);
98  QStyle *style = opt.widget ? opt.widget->style() : QApplication::style();
99  const int textMargin = style->pixelMetric(QStyle::PM_FocusFrameHMargin) + 1;
100  // QRect r = QStyle::alignedRect(opt.direction, Qt::AlignVCenter | Qt::AlignLeft, opt.decorationSize, r1);
101 
102  style->drawPrimitive(QStyle::PE_PanelItemViewItem, &opt, painter, opt.widget);
103  if (option.state & QStyle::State_Selected) {
104  painter->setPen(option.palette.highlightedText().color());
105  } else {
106  painter->setPen(option.palette.text().color());
107  }
108  QRect r = r1;
109  QFont font = painter->font();
110  font.setBold(true);
111  painter->setFont(font);
112  int decoWidth = int(2 * textMargin + r1.height() * 1.8);
113  r.setWidth(int(r1.height() * 1.8));
114  // Draw thumbnail
115  opt.icon.paint(painter, r);
116  int mid = r1.height() / 2;
117  r1.adjust(decoWidth, 0, 0, -mid);
118  QRect r2 = option.rect;
119  r2.adjust(decoWidth, mid, 0, 0);
120  QRectF bounding;
121  painter->drawText(r1, Qt::AlignLeft | Qt::AlignTop, index.data(Qt::DisplayRole).toString(), &bounding);
122  font.setBold(false);
123  painter->setFont(font);
124  QString subText = index.data(Qt::UserRole + 1).toString();
125  r2.adjust(0, int(bounding.bottom() - r2.top()), 0, 0);
126  QColor subTextColor = painter->pen().color();
127  subTextColor.setAlphaF(.5);
128  painter->setPen(subTextColor);
129  painter->drawText(r2, Qt::AlignLeft | Qt::AlignTop, subText, &bounding);
130  painter->restore();
131  } else {
132  QStyledItemDelegate::paint(painter, option, index);
133  }
