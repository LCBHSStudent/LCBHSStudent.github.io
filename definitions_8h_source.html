326 class enable_shared_from_this_virtual_base : public std::enable_shared_from_this<enable_shared_from_this_virtual_base>
327 {
328  using base_type = std::enable_shared_from_this<enable_shared_from_this_virtual_base>;
329  template <typename T> friend struct enable_shared_from_this_virtual;
330 
331  std::shared_ptr<enable_shared_from_this_virtual_base> shared_from_this() { return base_type::shared_from_this(); }
332  std::shared_ptr<enable_shared_from_this_virtual_base const> shared_from_this() const { return base_type::shared_from_this(); }
333 };
334 
335 template <typename T> struct enable_shared_from_this_virtual : virtual enable_shared_from_this_virtual_base
336 {
337  using base_type = enable_shared_from_this_virtual_base;
338 
339 public:
340  std::shared_ptr<T> shared_from_this()
341  {
342  std::shared_ptr<T> result(base_type::shared_from_this(), static_cast<T *>(this));
343  return result;
344  }
345 
346  std::shared_ptr<T const> shared_from_this() const
347  {
348  std::shared_ptr<T const> result(base_type::shared_from_this(), static_cast<T const *>(this));
349  return result;
350  }

353 // This is a small trick to have a QAbstractItemModel with shared_from_this enabled without multiple inheritance
354 // Be careful, if you use this class, you have to make sure to init weak_this_ when you construct a shared_ptr to your object
355 template <class T> class QAbstractItemModel_shared_from_this : public QAbstractItemModel
356 {
357 protected:
358  QAbstractItemModel_shared_from_this()
359  : QAbstractItemModel()
360  {
361  }
362 
363 public:
364  std::shared_ptr<T> shared_from_this()
365  {
366  std::shared_ptr<T> p(weak_this_);
367  assert(p.get() == this);
368  return p;
369  }
370 
371  std::shared_ptr<T const> shared_from_this() const
372  {
373  std::shared_ptr<T const> p(weak_this_);
374  assert(p.get() == this);

376  }
