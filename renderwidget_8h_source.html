52  void paint(QPainter *painter, const QStyleOptionViewItem &option, const QModelIndex &index) const override
53  {
54  if (index.column() == 1) {
55  painter->save();
56  QStyleOptionViewItem opt(option);
57  QStyle *style = opt.widget ? opt.widget->style() : QApplication::style();
58  const int textMargin = style->pixelMetric(QStyle::PM_FocusFrameHMargin) + 1;
59  style->drawPrimitive(QStyle::PE_PanelItemViewItem, &opt, painter, opt.widget);
60 
61  QFont font = painter->font();
62  font.setBold(true);
63  painter->setFont(font);
64  QRect r1 = option.rect;
65  r1.adjust(0, textMargin, 0, -textMargin);
66  int mid = int((r1.height() / 2));
67  r1.setBottom(r1.y() + mid);
68  QRect bounding;
69  painter->drawText(r1, Qt::AlignLeft | Qt::AlignTop, index.data().toString(), &bounding);
70  r1.moveTop(r1.bottom() - textMargin);
71  font.setBold(false);
72  painter->setFont(font);
73  painter->drawText(r1, Qt::AlignLeft | Qt::AlignTop, index.data(Qt::UserRole).toString());
74  int progress = index.data(Qt::UserRole + 3).toInt();
75  if (progress > 0 && progress < 100) {
76  // draw progress bar
77  QColor color = option.palette.alternateBase().color();
78  QColor fgColor = option.palette.text().color();
79  color.setAlpha(150);
80  fgColor.setAlpha(150);
81  painter->setBrush(QBrush(color));
82  painter->setPen(QPen(fgColor));
83  int width = qMin(200, r1.width() - 4);
84  QRect bgrect(r1.left() + 2, option.rect.bottom() - 6 - textMargin, width, 6);
85  painter->drawRoundedRect(bgrect, 3, 3);
86  painter->setBrush(QBrush(fgColor));
87  bgrect.adjust(2, 2, 0, -1);
88  painter->setPen(Qt::NoPen);
89  bgrect.setWidth((width - 2) * progress / 100);
90  painter->drawRect(bgrect);
91  } else {
92  r1.setBottom(opt.rect.bottom());
93  r1.setTop(r1.bottom() - mid);
94  painter->drawText(r1, Qt::AlignLeft | Qt::AlignBottom, index.data(Qt::UserRole + 5).toString());
95  }
96  painter->restore();
97  } else {
98  QStyledItemDelegate::paint(painter, option, index);
99  }
100  }
101 };
